public class Solution
{
    public string ToGoatLatin(string sentence)
    {
        var vowels = "aeiouAEIOU";
        string[] split = sentence.Split(" ");
        for (int i = 0; i < split.Length; i++)
        {
            if (vowels.Contains(split[i][0]))
            {
                split[i] += "ma";
            }
            else
            {

                var consonant = split[i][0];
                split[i] = split[i].Substring(1, split[i].Length - 1) + consonant + "ma";
            }

            StringBuilder strbldr = new StringBuilder(split[i]);
            int acount = 0;
            while (acount <= i)
            {
                strbldr.Append("a");
                acount++;
            }

            split[i] = strbldr.ToString();
        }

        var result = String.Join(" ", split);
        return result;
    }
}
